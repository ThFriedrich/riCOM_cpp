93 void run_connection_script(Ricom *ricom, MerlinSettings *merlin, std::string python_path)
94 {
95 
96  int m_fr_total = ((ricom->nx + ricom->skip_row) * ricom->ny + ricom->skip_img) * ricom->rep;
97 
98  std::ofstream run_script("run_script.py");
99  run_script << "from merlin_interface.merlin_interface import MerlinInterface" << std::endl;
100  run_script << "m = MerlinInterface(tcp_ip = \"" << ricom->socket.ip << "\" , tcp_port=" << merlin->com_port << ")" << std::endl;
101  run_script << "m.hvbias = " << merlin->hvbias << std::endl;
102  run_script << "m.threshold0 = " << merlin->threshold0 << std::endl;
103  run_script << "m.threshold1 = " << merlin->threshold1 << std::endl;
104  run_script << "m.continuousrw = " << (int)merlin->continuousrw << std::endl;
105  run_script << "m.counterdepth = " << ricom->camera.depth << std::endl;
106  run_script << "m.acquisitiontime = " << merlin->dwell_time << std::endl;
107  run_script << "m.acquisitionperiod = " << merlin->dwell_time << std::endl;
108  run_script << "m.numframestoacquire = " << m_fr_total + 1 << std::endl; // Ich verstehe nicht warum, aber er werkt.
109  run_script << "m.fileenable = " << (int)merlin->save << std::endl;
110  run_script << "m.runheadless = " << (int)merlin->headless << std::endl;
111  run_script << "m.fileformat = " << (int)merlin->raw * 2 << std::endl;
112  run_script << "m.triggerstart = " << merlin->trigger << std::endl;
113  run_script << "m.startacquisition()";
114  run_script.close();
115 
116  std::string command = python_path + " run_script.py";
117  int r = std::system(command.c_str());
118  if (r != 0)
119  {
120  std::cout << "main::run_connection_script: Cannot execute python run_script. Shell exited with code " << r << std::endl;
121  }
122 }

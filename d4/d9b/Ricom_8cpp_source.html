20 // Compute the kernel
21 void Ricom_kernel::compute_kernel()
22 {
23  float rot_rad = M_PI * rotation / 180;
24  float cos_rot = cos(rot_rad);
25  float sin_rot = sin(rot_rad);
26 
27  k_width_sym = kernel_size * 2 + 1;
28  k_area = k_width_sym * k_width_sym;
29  kernel_x.assign(k_area, 0);
30  kernel_y.assign(k_area, 0);
31  float d;
32  float ix_sd;
33  float iy_sd;
34  int ix_s;
35  int iy_s;
36  int ix_e;
37  int iy_e;
38 
39  for (int iy = 0; iy < k_width_sym; iy++)
40  {
41  iy_e = (iy + 1) * k_width_sym - 1;
42  for (int ix = 0; ix < k_width_sym; ix++)
43  {
44  ix_s = ix - kernel_size;
45  iy_s = iy - kernel_size;
46  d = ix_s * ix_s + iy_s * iy_s;
47 
48  ix_e = k_area - iy_e + ix - 1;
49 
50  if (d > 0)
51  {
52  ix_sd = (ix_s / d);
53  iy_sd = (iy_s / d);
54  kernel_x[ix_e] = cos_rot * ix_sd - sin_rot * iy_sd;
55  kernel_y[ix_e] = sin_rot * ix_sd + cos_rot * iy_sd;
56  }
57  else
58  {
59  kernel_y[ix_e] = 0;
60  kernel_x[ix_e] = 0;
61  }
62  }
63  }
64 
65  // Add filter
66  if (b_filter)
67  {
68  compute_filter();
69  include_filter();
70  }
71 }

73 // Compute the filter
74 void Ricom_kernel::compute_filter()
75 {
76  int filter_size = kernel_size * 2 + 1;
77  int dist = 0;
78  int ic = 0;
79  kernel_filter.assign(filter_size * filter_size, 0);
80 
81  for (int iy = 0; iy < filter_size; iy++)
82  {
83  for (int ix = 0; ix < filter_size; ix++)
84  {
85  dist = pow(ix - kernel_size, 2) + pow(iy - kernel_size, 2);
86  ic = iy * filter_size + ix;
87  if (dist <= pow(kernel_filter_frequency[0], 2) && dist > pow(kernel_filter_frequency[1], 2))
88  {
89  kernel_filter[ic] = 1;
90  }
91  }
92  }
93 }

95 // Applies the filter to the kernel
96 void Ricom_kernel::include_filter()
97 {
98  int k_width_sym = kernel_size * 2 + 1;
99  std::vector<int> map = fftshift_map(k_width_sym, k_width_sym);
100  std::complex<float> *k_x = new std::complex<float>[k_area];
101  std::complex<float> *k_y = new std::complex<float>[k_area];
102  std::complex<float> *k_x_f = new std::complex<float>[k_area];
103  std::complex<float> *k_y_f = new std::complex<float>[k_area];
104 
105  fftwf_plan px, py, ipx, ipy;
106 
107  for (int id = 0; id < k_area; id++)
108  {
109  k_x[id] = {kernel_x[id + map[id]], 0};
110  k_y[id] = {kernel_y[id + map[id]], 0};
111  }
112 
113  px = fftwf_plan_dft_2d(k_width_sym, k_width_sym, reinterpret_cast<fftwf_complex *>(k_x),
114  reinterpret_cast<fftwf_complex *>(k_x_f), FFTW_FORWARD, FFTW_ESTIMATE);
115  fftwf_execute(px);
116  fftwf_destroy_plan(px);
117 
118  py = fftwf_plan_dft_2d(k_width_sym, k_width_sym, reinterpret_cast<fftwf_complex *>(k_y),
119  reinterpret_cast<fftwf_complex *>(k_y_f), FFTW_FORWARD, FFTW_ESTIMATE);
120  fftwf_execute(py);
121  fftwf_destroy_plan(py);
122 
123  for (int id = 0; id < k_area; id++)
124  {
125  if (kernel_filter[id + map[id]] == 0)
126  {
127  k_x_f[id] = {0, 0};
128  k_y_f[id] = {0, 0};
129  }
130  }
131 
132  ipx = fftwf_plan_dft_2d(k_width_sym, k_width_sym, reinterpret_cast<fftwf_complex *>(k_x_f),
133  reinterpret_cast<fftwf_complex *>(k_x), FFTW_BACKWARD, FFTW_ESTIMATE);
134  fftwf_execute(ipx);
135  fftwf_destroy_plan(ipx);
136 
137  ipy = fftwf_plan_dft_2d(k_width_sym, k_width_sym, reinterpret_cast<fftwf_complex *>(k_y_f),
138  reinterpret_cast<fftwf_complex *>(k_y), FFTW_BACKWARD, FFTW_ESTIMATE);
139  fftwf_execute(ipy);
140  fftwf_destroy_plan(ipy);
141 
142  for (int id = 0; id < k_area; id++)
143  {
144  kernel_x[id + map[id]] = k_x[id].real();
145  kernel_y[id + map[id]] = k_y[id].real();
146  }
147  delete[] k_x;
148  delete[] k_y;
149  delete[] k_x_f;
150  delete[] k_y_f;
151 }

153 // FFTSHIFT implementation
154 std::vector<int> Ricom_kernel::fftshift_map(int x, int y)
155 {
156  int center_x = round(x / 2);
157  int center_y = round(y / 2);
158  std::vector<int> map(x * y);
159 
160  int shift_x, shift_y;
161  if (x % 2 == 0)
162  {
163  shift_x = 0;
164  }
165  else
166  {
167  shift_x = -1;
168  }
169  if (y % 2 == 0)
170  {
171  shift_y = 0;
172  }
173  else
174  {
175  shift_y = -1;
176  }
177 
178  int q1 = center_y * x + center_x;
179  int q2 = center_y * x - (center_x + shift_x);
180  int q3 = -1 * (center_y + shift_y) * x + center_x;
181  int q4 = -1 * (center_y + shift_y) * x - (center_x + shift_x);
182 
183  int cnt = 0;
184  for (int iy = 0; iy < center_y; iy++)
185  {
186  for (int ix = 0; ix < center_x; ix++)
187  {
188  map[cnt] = q1;
189  cnt++;
190  }
191  for (int ix = center_x; ix < x; ix++)
192  {
193  map[cnt] = q2;
194  cnt++;
195  }
196  }
197 
198  for (int iy = center_y; iy < y; iy++)
199  {
200  for (int ix = 0; ix < center_x; ix++)
201  {
202  map[cnt] = q3;
203  cnt++;
204  }
205  for (int ix = center_x; ix < x; ix++)
206  {
207  map[cnt] = q4;
208  cnt++;
209  }
210  }
211  return map;
212 }

214 void Ricom_kernel::approximate_frequencies(size_t nx_im)
215 {
216  f_approx.resize(nx_im);
217  float f_max = 0;
218  float k = kernel_size * 2;
219  for (size_t i = 0; i < nx_im; i++)
220  {
221  float x = 2 * i * M_PI;
222  f_approx[i] = (nx_im / x) * (1 - cos(k / 2 * (x / nx_im)));
223  if (f_approx[i] > f_max)
224  {
225  f_max = f_approx[i];
226  }
227  }
228  for (float &f : f_approx)
229  {
230  f /= f_max;
231  }
232 }

399 // Change the endianness of the incoming data
400 template <typename T>
401 void Ricom::swap_endianess(T &val)
402 {
403  if (val > 0 && camera.swap_endian)
404  {
405  switch (sizeof(T))
406  {
407  case 2:
408  val = (val >> 8) | ((val & 0xff) << 8);
409  break;
410  case 4:
411  val = (val >> 24) | ((val & 0xff) << 24) | ((val & 0xff00) << 8) | ((val & 0xff0000) >> 8);
412  default:
413  break;
414  }
415  }
416 }

418 // Compute the centre of mass
419 template <typename T>
420 void Ricom::com(std::vector<T> *data, std::array<float, 2> &com, std::atomic<int> *dose_sum)
421 {
422  float dose = 0;
423  T px;
424  size_t sum_x_temp;
425  std::vector<size_t> sum_x(camera.nx_cam);
426  std::vector<size_t> sum_y(camera.ny_cam);
427  sum_x.assign(camera.nx_cam, 0);
428  sum_y.assign(camera.ny_cam, 0);
429  com = {0.0, 0.0};
430 
431  size_t y_nx;
432  for (int idy = 0; idy < camera.ny_cam; idy++)
433  {
434  y_nx = idy * camera.nx_cam;
435  sum_x_temp = 0;
436  for (int idx = 0; idx < camera.nx_cam; idx++)
437  {
438  px = data->data()[y_nx + idx];
439  swap_endianess(px);
440  sum_x_temp += px;
441  sum_y[idx] += px;
442  }
443  sum_x[idy] = sum_x_temp;
444  dose += sum_x_temp;
445  }
446 
447  if (dose > 0)
448  {
449  for (int i = 0; i < camera.nx_cam; i++)
450  {
451  com[0] += sum_x[i] * camera.v[i];
452  }
453  for (int i = 0; i < camera.ny_cam; i++)
454  {
455  com[1] += sum_y[i] * camera.u[i];
456  }
457  for (int i = 0; i < 2; i++)
458  {
459  com[i] = (com[i] / dose);
460  }
461  dose_sum[0] += (int)dose;
462  }
463 }

466 template <typename T>
467 void Ricom::stem(std::vector<T> *data, size_t id_stem)
468 {
469  T px;
470  size_t stem_temp = 0;
471  for (size_t id : detector.id_list)

486  rescale_stem = true;
487  }
488  }
489  }
490  stem_data[id_stem] = stem_temp;

625 // Draw a CBED in log-scale to the SDL surface srf_cbed
626 template <typename T>
627 void Ricom::plot_cbed(std::vector<T> *cbed_data)
628 {
629  float v_min = INFINITY;
630  float v_max = 0.0;
631  float vl_f;
632 
633  for (size_t id = 0; id < (*cbed_data).size(); id++)
634  {
635  T vl = (*cbed_data)[id];
636  swap_endianess(vl);
637  vl_f = log(1.0 + (float)vl);
638  if (vl_f > v_max)
639  {
640  v_max = vl_f;
641  }
642  if (vl_f < v_min)
643  {
644  v_min = vl_f;
645  }
646  cbed_log[id] = vl_f;
647  }
648 
649  int iy_t;
650  float v_rng = v_max - v_min;
651  for (int ix = 0; ix < camera.ny_cam; ix++)
652  {
653  iy_t = camera.v[ix] * camera.nx_cam;

779 // Process FRAME_BASED camera data
780 template <typename T, class CameraInterface>
781 void Ricom::process_data(CAMERA::Camera<CameraInterface, CAMERA::FRAME_BASED> *camera_spec)
782 {
783  // Memory allocation
784  std::vector<T> data(camera_spec->nx_cam * camera_spec->ny_cam);
785  std::vector<T> *p_data = &data;
786  BoundedThreadPool pool;
787 
788  // Start Thread Pool
789  if (n_threads > 1)
790  pool.init(n_threads, queue_size);
791 
792  std::atomic<int> dose_sum = 0;
793  std::atomic<int> *p_dose_sum = &dose_sum;
794 
795  std::array<float, 2> com_xy_sum = {0.0, 0.0};
796  std::array<float, 2> *p_com_xy_sum = &com_xy_sum;
797 
798  // Initialize ProgressMonitor Object
799  ProgressMonitor prog_mon(fr_total, !b_print2file, redraw_interval);
800  p_prog_mon = &prog_mon;
801 
802  for (int ir = 0; ir < rep; ir++)
803  {
804  reinit_vectors_limits();
805  for (int iy = 0; iy < ny; iy++)
806  {
807  for (int ix = 0; ix < nx; ix++)
808  {
809  camera_spec->read_frame(data, !p_prog_mon->first_frame);
810  p_prog_mon->first_frame = false;
811  if (n_threads > 1)
812  {
813  pool.push_task([=]
814  { com_icom<T>(data, ix, iy, p_dose_sum, p_com_xy_sum, p_prog_mon); });
815  }
816  else
817  {
818  com_icom<T>(p_data, ix, iy, p_dose_sum, p_com_xy_sum, p_prog_mon);
819  }
820 
821  if (rc_quit)
822  {
823  pool.wait_for_completion();
824  p_prog_mon = nullptr;
825  return;
826  };
827  }
828  skip_frames(skip_row, data, camera_spec);
829  }
830  skip_frames(skip_img, data, camera_spec);
831 
832  if (n_threads > 1)
833  pool.wait_for_completion();
834 
835  if (update_offset && dose_sum > pow(10.0, update_dose_lowbound))
836  {
837  offset[0] = com_public[0];
838  offset[1] = com_public[1];
839  dose_sum = 0;
840  }
841  }
842  p_prog_mon = nullptr;

861 // Process EVENT_BASED camera data
862 template <class CameraInterface>
863 void Ricom::process_data(CAMERA::Camera<CameraInterface, CAMERA::EVENT_BASED> *camera_spec)
864 {
865  // Memory allocation
866  std::vector<size_t> dose_map(nxy);
867  std::vector<size_t> sumx_map(nxy);
868  std::vector<size_t> sumy_map(nxy);
869  std::vector<uint16_t> frame(camera_spec->nx_cam * camera_spec->ny_cam);
870  std::vector<uint16_t> *p_frame = &frame;
871 
872  BoundedThreadPool pool;
873 
874  // Start Thread Pool
875  if (n_threads > 1)
876  pool.init(n_threads, queue_size);
877 
878  std::array<float, 2> com_xy = {0.0, 0.0};
879  std::array<float, 2> *p_com_xy = &com_xy;
880  std::array<float, 2> com_xy_sum = {0.0, 0.0};
881  int dose_sum = 0;
882  int idxx = 0;
883  int ix = 0;
884  int iy = 0;
885  int acc_cbed = 0;
886  int acc_idx = 0;
887 
888  size_t img_num = 0;
889  size_t first_frame = img_num * nxy;
890  size_t end_frame = (img_num + 1) * nxy;
891  size_t fr_total_u = (size_t)fr_total;
892 
893  ProgressMonitor prog_mon(fr_total, !b_print2file, redraw_interval);
894  p_prog_mon = &prog_mon;
895 
896  dose_map.assign(nxy, 0);
897  sumx_map.assign(nxy, 0);
898  sumy_map.assign(nxy, 0);
899  reinit_vectors_limits();
900 
901  while (true)
902  {
903  // process two frames before the probe position to avoid toa problem
904  idxx = prog_mon.fr_count - nxy * img_num - 2;
905 
906  ++prog_mon;
907  fr_count = prog_mon.fr_count;
908 
909  if (acc_cbed < 3 && b_plot_cbed)
910  {
911  camera_spec->read_frame_com_cbed(prog_mon.fr_count,
912  dose_map, sumx_map, sumy_map,
913  stem_data, b_vSTEM,
914  offset, detector.radius2,
915  frame, acc_idx,
916  first_frame, end_frame);
917  acc_cbed +=1;
918  }
919  else
920  {
921  camera_spec->read_frame_com(prog_mon.fr_count,
922  dose_map, sumx_map, sumy_map,
923  stem_data, b_vSTEM,
924  offset, detector.radius2,
925  first_frame, end_frame);
926  }
927 
928  if (idxx >= 0)
929  {
930  if (dose_map[idxx] == 0)
931  {
932  com_xy[0] = offset[0];
933  com_xy[1] = offset[1];
934  }
935  else
936  {
937  com_xy[0] = sumx_map[idxx] / dose_map[idxx];
938  com_xy[1] = sumy_map[idxx] / dose_map[idxx];
939  }
940  com_map_x[idxx] = com_xy[0];
941  com_map_y[idxx] = com_xy[1];
942  com_xy_sum[0] += com_xy[0];
943  com_xy_sum[1] += com_xy[1];
944  dose_sum += dose_map[idxx];
945 
946  ix = idxx % nx;
947  iy = floor(idxx / nx);
948 
949  if (n_threads > 1)
950  {
951  pool.push_task([=]
952  { icom(com_xy, ix, iy); });
953  }
954  else
955  {
956  icom(p_com_xy, ix, iy);
957  }
958 
959  // if (b_vSTEM)
960  // {
961  // stem(p_frame, iy * nx + ix);
962  // }
963  }
964 
965  if (prog_mon.report_set)
966  {
967  update_surfaces(iy, p_frame);
968  if (b_plot_cbed)
969  {
970  frame.assign(camera_spec->nx_cam * camera_spec->ny_cam, 0);
971  acc_cbed = 0;
972  acc_idx = fr_count;
973  }
974  fr_freq = prog_mon.fr_freq;
975  rescales_recomputes();
976  for (int i = 0; i < 2; i++)
977  {
978  com_public[i] = com_xy_sum[i] / prog_mon.fr_count_i;
979  com_xy_sum[i] = 0;
980  }
981  prog_mon.reset_flags();
982  last_y = iy;
983  }
984 
985  if (prog_mon.fr_count >= end_frame)
986  {
987  if (prog_mon.fr_count != fr_total_u)
988  {
989  img_num++;
990  first_frame = img_num * nxy;
991  end_frame = (img_num + 1) * nxy;
992  reinit_vectors_limits();
993  dose_map.assign(nxy, 0);
994  sumx_map.assign(nxy, 0);
995  sumy_map.assign(nxy, 0);
996  }
997 
998  if (update_offset && dose_sum > pow(10.0, update_dose_lowbound))
999  {
1000  offset[0] = com_public[0];
1001  offset[1] = com_public[1];
1002  dose_sum = 0;
1003  }
1004  }
1005 
1006  if (prog_mon.fr_count == fr_total_u || rc_quit)
1007  {
1008  pool.wait_for_completion();
1009  p_prog_mon = nullptr;
1010  return;
1011  }
1012 }
1013 p_prog_mon = nullptr;
1014 }

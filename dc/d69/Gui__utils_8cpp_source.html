82 {

91 void Generic_Image_Window::set_min_max(int last_idr)
92 {
93  float val;
94  for (int idr = this->last_idr; idr < last_idr; idr++)
95  {
96  val = (*data)[idr];
97  if (val < data_min)
98  {
99  data_min = val;
100  data_range = data_max - data_min;
101  }
102  if (val > data_max)
103  {
104  data_max = val;
105  data_range = data_max - data_min;
106  }
107  }
108 }

347  float rel_y = xy.y - pos.y - ImGui::GetScrollY();

368  {

370  zoom = (std::max)(1.0f, zoom);
371  ImGui::SetScrollX(start_x * (zoom - 1.0f));

377  {
378  float scale_fct = scale * zoom;
379  int x = (int)std::floor((xy.x - pos.x) / scale_fct);
380  int y = (int)std::floor((xy.y - pos.y) / scale_fct);

382  ImGui::BeginTooltip();
383  ImGui::Text("XY: %i, %i", x, y);
384  ImGui::Text("Value: %.2f", val);
385  ImGui::Text("Zoom: %.1f", zoom);
386  ImGui::EndTooltip();
387  }
388  }
389  ImGui::EndChildFrame();
390  }
391  ImGui::End();
392  b_trigger_update = false;
393 }
